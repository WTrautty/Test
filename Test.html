=== Step 1: Get user input output_file_path = input("Enter full path for the output .txt file (e.g., C:\\output\\report.txt): ").strip() template_path = input("Enter full path for the .txt template file (e.g., C:\\templates\\template.txt): ").strip() order_number = input("Enter the order number: ").strip() # === Utility: Find header row def find_header_row(sheet): data = np.array([row for row in sheet.iter_rows(min_row=1, max_row=20, values_only=True)], dtype=object) for row_idx, row in enumerate(data, start=1): if np.count_nonzero(row) > 2: return [str(cell).strip() if cell else "" for cell in row], row_idx return None, None # === Search Lumetions Tracker === def search_order(sheet, order_num): header_row, header_row_idx = find_header_row(sheet) if not header_row: return None headers = {cell.replace('\n', ' ').replace('\r', '').strip(): idx for idx, cell in enumerate(header_row) if cell} search_cols = [col for col in headers if any(col.strip().startswith(s) for s in SEARCH_COLUMNS)] extract_cols = {} for extract_key in EXTRACT_COLUMNS: for header_key in headers: if header_key.strip().startswith(extract_key.strip()): extract_cols[extract_key] = headers[header_key] break for row in sheet.iter_rows(min_row=header_row_idx + 1, values_only=True): for col in search_cols: col_idx = headers[col] cell_value = str(row[col_idx]) if row[col_idx] else "" if order_num in cell_value: return {col.strip(): row[idx] for col, idx in extract_cols.items()} return None # === Search LASER Export for ALL matching Service IDs === def search_laser_by_service_id(service_id): try: laser_wb = openpyxl.load_workbook(LASER_FILE, read_only=True) except FileNotFoundError: print(f"‚ùå Error: LASER file not found - {LASER_FILE}") return [] if LASER_SHEET not in laser_wb.sheetnames: print(f"‚ùå Error: Sheet '{LASER_SHEET}' not found in LASER file.") return [] ws = laser_wb[LASER_SHEET] header_row, header_row_idx = find_header_row(ws) if not header_row: return [] headers = {cell.replace('\n', ' ').replace('\r', '').strip(): idx for idx, cell in enumerate(header_row) if cell} match_col = None for key in headers: if key.lower().startswith(LASER_MATCH_COLUMN.lower()): match_col = headers[key] break extract_cols = {} for extract_key in LASER_EXTRACT_COLUMNS: for header_key in headers: if header_key.lower().startswith(extract_key.lower()): extract_cols[extract_key] = headers[header_key] break matches = [] for row in ws.iter_rows(min_row=header_row_idx + 1, values_only=True): cell_value = str(row[match_col]) if row[match_col] else "" if service_id in cell_value: match = {col: row[idx] for col, idx in extract_cols.items()} matches.append(match) return matches # === Load Lumetions workbook and search try: print(f"\nüìÇ Loading Lumetions Tracker: {EXCEL_FILE}") wb = openpyxl.load_workbook(EXCEL_FILE, read_only=True) except FileNotFoundError: print(f"‚ùå Error: File not found - {EXCEL_FILE}") exit() if SHEET_NAME not in wb.sheetnames: print(f"‚ùå Error: Sheet '{SHEET_NAME}' not found.") wb.close() exit() result_data = search_order(wb[SHEET_NAME], order_number) wb.close() # === Load template try: with open(template_path, "r", encoding="utf-8") as f: template = f.read() print("‚úÖ Template loaded.\n") except FileNotFoundError: print(f"‚ùå Error: Template file '{template_path}' not found.") exit() # === Fill and export if result_data: service_id = str(result_data.get("Service ID", "")).strip() laser_matches = search_laser_by_service_id(service_id) if not laser_matches: laser_matches = [{}] # fallback: no matches, but still output 1 result for i, laser_data in enumerate(laser_matches, start=1): filled_template = template for key in EXTRACT_COLUMNS: placeholder = f"{{{key.strip()}}}" filled_template = filled_template.replace(placeholder, str(result_data.get(key.strip(), ""))) for key in LASER_EXTRACT_COLUMNS: placeholder = f"{{{key}}}" filled_template = filled_template.replace(placeholder, str(laser_data.get(key, ""))) filled_template = filled_template.replace("{OrderNumber}", order_number) # Handle output file naming output_path = output_file_path if len(laser_matches) == 1 else output_file_path.replace(".txt", f"_{i}.txt") try: with open(output_path, "w", encoding="utf-8") as out_file: out_file.write(filled_template) print(f"‚úÖ Output written to: {output_path}") except Exception as e: print(f"‚ùå Failed to write to output file: {e}") else: print(f"‚ùå Order number '{order_number}' not found in Lumetions Tracker.")
