import glob import os import re import ipaddress txt_files = glob.glob(os.path.join(txt_folder_path, '**', '*.txt'), recursive=True) print(f"Checking {len(txt_files)} TXT files...") for txt_path in txt_files: try: with open(txt_path, 'r', encoding='utf-8', errors='ignore') as f: content = f.read().lower() if '9200' in os.path.basename(txt_path) and 'wr mem' not in content: failed_files.append(f"{os.path.basename(txt_path)}\n{txt_path}\nMissing 'wr mem' in 9200 TXT file.\n") if 'commit' not in content: failed_files.append(f"{os.path.basename(txt_path)}\n{txt_path}\nMissing 'commit' in TXT file.\n") # Find lines with .254 and IP address pattern f.seek(0) lines = f.readlines() ip_lines = [] ip_pattern = re.compile(r'ip address (\d+\.\d+\.\d+\.\d+)\s+(\d+\.\d+\.\d+\.\d+)') for line in lines: if '.254' in line: match = ip_pattern.search(line) if match: ip_lines.append((match.group(1), match.group(2))) # (IP, Subnet) if len(ip_lines) == 2: break if len(ip_lines) == 2: ip1, subnet1 = ip_lines[0] ip2, subnet2 = ip_lines[1] try: ip1_obj = ipaddress.IPv4Address(ip1) ip2_obj = ipaddress.IPv4Address(ip2) if subnet1 != subnet2: failed_files.append(f"{os.path.basename(txt_path)}\n{txt_path}\nSubnet mismatch between two .254 lines: {subnet1} vs {subnet2}\n") elif int(ip2_obj) != int(ip1_obj) + 1: failed_files.append(f"{os.path.basename(txt_path)}\n{txt_path}\nSecond .254 IP ({ip2}) is not one higher than first ({ip1})\n") except ValueError: failed_files.append(f"{os.path.basename(txt_path)}\n{txt_path}\nInvalid IP address format in .254 check.\n") else: failed_files.append(f"{os.path.basename(txt_path)}\n{txt_path}\nLess than two valid '.254' IP address lines found.\n") except Exception as e: failed_files.append(f"Read Error: {os.path.basename(txt_path)}\n{txt_path}\nError: {e}\n")
