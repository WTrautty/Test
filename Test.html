import os import re import shutil import glob def process_file(filepath): with open(filepath, 'r') as file: content = file.read() # Use a regex to extract text between the delimiters. pattern = re.compile(r'AUTOMATED COMMANDS OUTPUT:(.*?)END OF AUTOMATED OUTPUT', re.DOTALL) match = pattern.search(content) if not match: print(f"Delimiters not found in {filepath}") return section = match.group(1) base_dir = os.path.dirname(filepath) filename = os.path.basename(filepath) # Check for conditions and create target folder and new filename accordingly. if "0 packets received" in section: target_folder = os.path.join(base_dir, "unreachable") new_filename = "Unpingable " + filename elif "/126" in section or "/56" in section: target_folder = os.path.join(base_dir, "Routes Found") new_filename = "Routes Found " + filename else: print(f"No conditions met for {filename}") return # Create the target folder if it doesn't exist. os.makedirs(target_folder, exist_ok=True) # Construct the new file path. target_path = os.path.join(target_folder, new_filename) # Move (and rename) the file. shutil.move(filepath, target_path) print(f"Moved {filename} to {target_folder} as {new_filename}") def main(): # Find all .txt files in the current directory. txt_files = glob.glob("*.txt") for filepath in txt_files: process_file(filepath) if __name__ == '__main__': main()
