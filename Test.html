def load_commands_from_file(file_path: str, delimiter: str = "\n\n") -> List[str]: """ Reads a file and splits its content into commands based on the given delimiter. Args: file_path: The path to the file containing commands. delimiter: The delimiter used to separate commands (default is two newlines). Returns: A list of command strings. """ try: with open(file_path, 'r') as file: content = file.read().strip() # Split based on the delimiter and remove any empty strings commands = [cmd.strip() for cmd in content.split(delimiter) if cmd.strip()] return commands except IOError as e: print(f"Error reading {file_path}: {e}") return []
